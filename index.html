// 1. دي الفانكشن اللي بتعرض الأفلام في الصفحة function renderMovies(moviesList) { const container = document.getElementById('movies-container'); // اتأكد إن ده نفس الـ ID عندك container.innerHTML = ''; // عشان يمسح القديم ويعرض الجديد moviesList.forEach(movie => { // تصليح تلقائي لروابط ImgBB لو مش مباشرة let finalUrl = movie.url; if (finalUrl.includes('ibb.co') && !finalUrl.match(/\.(jpg|jpeg|png|gif)$/)) { // محاولة بسيطة لتحويل الرابط لرابط مباشر لو نسيت console.warn("انتبه: الرابط قد لا يكون مباشراً للفيلم:", movie.title); } const movieCard = `
${movie.title}
`; container.insertAdjacentHTML('beforeend', movieCard); }); }
